MySQL:
	Utilisateurs:
		id Integer primary key auto-increment
		pseudo Varchar(32) unique
		mot_de_passe Varchar(64) text
		mail Varchar(64)
		prenom Varchar(64)
		nom Varchar(64)
		anniversaire Date
				
	Amities:
		id_ami1 Integer
		id_ami2 Integer
		timestamp Timestamp 
		primary key (id_ami1, id_ami2)

	Sessions:
		clef Varchar(32) unique
		id Integer
		timestamp Timestamp
		est_administrateur boolean

MongoDB:
	Messages:
		id_auteur int
		contenu String
		timestamp Timestamp
		retweests [id1, id2, ...]
